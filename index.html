const elements = document.querySelectorAll('[data-fa]'); const btn = document.getElementById('langToggle'); let lang = localStorage.getItem('lang') || 'fa'; function setLang(language) { elements.forEach(el => { el.textContent = el.dataset[language]; }); document.documentElement.lang = language; document.body.dir = language === 'fa' ? 'rtl' : 'ltr'; localStorage.setItem('lang', language); } btn.onclick = () => { lang = lang === 'fa' ? 'en' : 'fa'; setLang(lang); }; setLang(lang);
